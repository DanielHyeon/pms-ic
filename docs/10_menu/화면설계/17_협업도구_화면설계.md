function meetingFilterSpecToUrl(
  filter: MeetingFilterSpec,
  entityId?: string,
  view?: string
): string {
  if (entityId) return `/meetings/${entityId}`;

  const params = new URLSearchParams();
  for (const [key, value] of Object.entries(filter)) {
    if (value !== undefined) {
      params.set(key, String(value));
    }
  }
  if (view) params.set("view", view);
  return `/meetings?${params.toString()}`;
}

function noticeFilterSpecToUrl(
  filter: NoticeFilterSpec,
  entityId?: string
): string {
  if (entityId) return `/notices/${entityId}`;

  const params = new URLSearchParams();
  for (const [key, value] of Object.entries(filter)) {
    if (value !== undefined) {
      params.set(key, String(value));
    }
  }
  return `/notices?${params.toString()}`;
}

### 12.1 ÌöåÏùò Í¥ÄÎ†®

| # | Risk Area | Problem | Prevention |
